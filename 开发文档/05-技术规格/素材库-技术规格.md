## 4. 数据库表结构

### 4.1 materials 表

```sql
CREATE TABLE materials (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name            VARCHAR(255) NOT NULL,                       -- 📥存储 素材名称
  type            VARCHAR(20) NOT NULL,                        -- 📥存储 枚举：图片/视频/音乐/音效/地图动画
  file_url        TEXT NOT NULL,                               -- 文件存储地址
  thumbnail_url   TEXT,                                        -- 📥存储 缩略图地址
  file_size       BIGINT NOT NULL DEFAULT 0,                   -- 📥存储 文件大小（字节）
  mime_type       VARCHAR(100),                               -- 文件MIME类型
  tags            JSONB DEFAULT '[]'::jsonb,                   -- 📥存储 标签数组
  source          VARCHAR(20) NOT NULL DEFAULT '自拍',          -- 📥存储 枚举：Pexels/AI生成/自拍/购买/外部导入
  copyright_status VARCHAR(20) NOT NULL DEFAULT '待确认',       -- 📥存储 枚举：免费商用/已购买/AI生成/待确认
  use_count       INTEGER NOT NULL DEFAULT 0,                  -- 📥存储 使用次数
  file_hash       VARCHAR(64),                                -- 文件哈希（去重用）
  metadata        JSONB DEFAULT '{}'::jsonb,                   -- 扩展元数据（宽高/时长/采样率等）
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  CONSTRAINT chk_material_type CHECK (type IN ('图片', '视频', '音乐', '音效', '地图动画')),
  CONSTRAINT chk_material_source CHECK (source IN ('Pexels', 'AI生成', '自拍', '购买', '外部导入')),
  CONSTRAINT chk_material_copyright CHECK (copyright_status IN ('免费商用', '已购买', 'AI生成', '待确认'))
);

-- 索引
CREATE INDEX idx_materials_type ON materials(type);
CREATE INDEX idx_materials_source ON materials(source);
CREATE INDEX idx_materials_created_at ON materials(created_at DESC);
CREATE INDEX idx_materials_use_count ON materials(use_count DESC);
CREATE INDEX idx_materials_name ON materials(name);
CREATE INDEX idx_materials_file_hash ON materials(file_hash);
CREATE INDEX idx_materials_tags ON materials USING GIN(tags);
```

## 5. 状态管理

### Pinia Store：useMaterialStore

文件路径：`src/stores/materialStore.ts`

```typescript
import { defineStore } from 'pinia'

export const useMaterialStore = defineStore('material', () => {
  // ===== 核心数据 =====
  /** 素材列表，溯源：数据流-前端状态 materials[] */
  const materials = ref<Material[]>([])

  /** 素材统计，溯源：数据流-前端状态 material_stats */
  const materialStats = ref<MaterialStats>({
    total: 0, image: 0, video: 0, music: 0, sfx: 0, map_animation: 0
  })

  // ===== 筛选状态 =====
  /** 类型筛选，溯源：数据流-前端状态 type_filter */
  const typeFilter = ref<MaterialType | null>(null)

  /** 排序方式，溯源：数据流-前端状态 sort_by */
  const sortBy = ref<MaterialSortOption>('created_at')

  /** 搜索关键词，溯源：数据流-前端状态 search_keyword */
  const searchKeyword = ref('')

  /** 当前页码，溯源：数据流-前端状态 current_page */
  const currentPage = ref(1)

  /** 总数（用于判断是否还有更多） */
  const total = ref(0)

  // ===== 加载状态 =====
  /** 列表加载中，溯源：数据流-前端状态 loading */
  const loading = ref(false)

  /** 上传中，溯源：数据流-前端状态 uploading */
  const uploading = ref(false)

  // ===== 弹窗可见性 =====
  /** 上传弹窗，溯源：数据流-前端状态 show_upload_modal */
  const showUploadModal = ref(false)

  /** AI生成弹窗，溯源：数据流-前端状态 show_ai_modal */
  const showAiModal = ref(false)

  /** 编辑弹窗，溯源：数据流-前端状态 show_edit_modal */
  const showEditModal = ref(false)

  /** 删除弹窗，溯源：数据流-前端状态 show_delete_modal */
  const showDeleteModal = ref(false)

  // ===== 编辑状态 =====
  /** 当前编辑的素材，溯源：数据流-前端状态 editing_material */
  const editingMaterial = ref<Material | null>(null)

  /** 待删除的素材ID */
  const deletingMaterialId = ref<string | null>(null)

  // ===== AI生成状态 =====
  /** AI生成类型，溯源：数据流-前端状态 gen_type */
  const genType = ref<'图片' | '视频'>('图片')

  /** AI生成描述，溯源：数据流-前端状态 gen_description */
  const genDescription = ref('')

  /** AI生成任务ID，溯源：数据流-前端状态 gen_task_id */
  const genTaskId = ref<string | null>(null)

  /** AI生成进度，溯源：数据流-前端状态 gen_progress */
  const genProgress = ref(0)

  // ===== 上传状态 =====
  /** 上传文件列表，溯源：数据流-前端状态 upload_files[] */
  const uploadFiles = ref<File[]>([])

  /** 上传进度Map，溯源：数据流-前端状态 upload_progress_map */
  const uploadProgressMap = ref<Map<string, number>>(new Map())

  // ===== 计算属性 =====
  /** 是否还有更多数据可加载 */
  const hasMore = computed(() => materials.value.length < total.value)

  /** 是否为空（初次加载后无数据） */
  const isEmpty = computed(() => !loading.value && materials.value.length === 0 && currentPage.value === 1)

  /** 是否搜索无结果 */
  const isSearchEmpty = computed(() => !loading.value && materials.value.length === 0 && (searchKeyword.value !== '' || typeFilter.value !== null))

  // ===== Actions =====

  /** 加载素材列表，溯源：数据流-状态流转"筛选/搜索状态流转" */
  async function fetchMaterials(resetList: boolean = false): Promise<void> {
    if (resetList) {
      currentPage.value = 1
      materials.value = []
    }
    loading.value = true
    try {
      const response = await materialService.getMaterials({
        type: typeFilter.value,
        sort_by: sortBy.value,
        keyword: searchKeyword.value || undefined,
        page: currentPage.value,
        page_size: 20
      })
      if (resetList) {
        materials.value = response.items
      } else {
        materials.value.push(...response.items)
      }
      total.value = response.total
      materialStats.value = response.stats
    } finally {
      loading.value = false
    }
  }

  /** 加载更多（滚动到底部触发） */
  async function loadMore(): Promise<void> {
    if (!hasMore.value || loading.value) return
    currentPage.value++
    await fetchMaterials(false)
  }

  /** 上传素材，溯源：数据流-状态流转"素材上传状态流转" */
  async function uploadMaterials(files: File[]): Promise<void> {
    uploading.value = true
    try {
      const result = await materialService.uploadMaterials(
        files,
        undefined,
        (fileIndex, progress) => {
          uploadProgressMap.value.set(files[fileIndex].name, progress)
        }
      )
      showUploadModal.value = false
      uploadFiles.value = []
      uploadProgressMap.value.clear()
      await fetchMaterials(true)
    } finally {
      uploading.value = false
    }
  }

  /** AI生成素材，溯源：数据流-状态流转"AI生成素材状态流转" */
  async function generateMaterial(): Promise<void> {
    const result = await materialService.generateMaterial({
      type: genType.value,
      description: genDescription.value
    })
    genTaskId.value = result.task_id
    // 开始轮询
    await pollGenerateStatus()
  }

  /** 轮询AI生成状态，2秒间隔 */
  async function pollGenerateStatus(): Promise<void> {
    if (!genTaskId.value) return
    const status = await materialService.getGenerateStatus(genTaskId.value)
    genProgress.value = status.progress
    if (status.status === 'generating') {
      setTimeout(() => pollGenerateStatus(), 2000)
    } else if (status.status === 'completed') {
      // 用户确认后入库
    }
  }

  /** 确认AI生成素材 */
  async function confirmGeneratedMaterial(): Promise<void> {
    if (!genTaskId.value) return
    await materialService.confirmGeneratedMaterial(genTaskId.value)
    genTaskId.value = null
    genProgress.value = 0
    genDescription.value = ''
    await fetchMaterials(true)
  }

  /** 编辑素材信息，溯源：数据流-状态流转"素材编辑状态流转" */
  async function updateMaterial(id: string, params: {
    name?: string; tags?: string[]; copyright_status?: CopyrightStatus; source?: MaterialSource
  }): Promise<void> {
    const result = await materialService.updateMaterial(id, params)
    // 更新本地列表中对应项
    const index = materials.value.findIndex(m => m.id === id)
    if (index !== -1) {
      materials.value[index] = result.material
    }
  }

  /** 删除素材，溯源：数据流-状态流转"素材删除状态流转" */
  async function deleteMaterial(id: string): Promise<void> {
    await materialService.deleteMaterial(id)
    materials.value = materials.value.filter(m => m.id !== id)
    await fetchMaterials(true)
  }

  /** 检查素材引用 */
  async function checkReferences(id: string): Promise<number> {
    const result = await materialService.getMaterialReferences(id)
    return result.count
  }

  /** 设置筛选条件并刷新 */
  function setFilter(filter: {
    type?: MaterialType | null
    sort?: MaterialSortOption
    keyword?: string
  }): void {
    if (filter.type !== undefined) typeFilter.value = filter.type
    if (filter.sort !== undefined) sortBy.value = filter.sort
    if (filter.keyword !== undefined) searchKeyword.value = filter.keyword
    fetchMaterials(true)
  }

  return {
    // state
    materials, materialStats, typeFilter, sortBy, searchKeyword,
    currentPage, total, loading, uploading,
    showUploadModal, showAiModal, showEditModal, showDeleteModal,
    editingMaterial, deletingMaterialId,
    genType, genDescription, genTaskId, genProgress,
    uploadFiles, uploadProgressMap,
    // computed
    hasMore, isEmpty, isSearchEmpty,
    // actions
    fetchMaterials, loadMore, uploadMaterials,
    generateMaterial, pollGenerateStatus, confirmGeneratedMaterial,
    updateMaterial, deleteMaterial, checkReferences, setFilter
  }
})
```
