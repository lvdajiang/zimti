## 5. 状态管理

### 5.1 Pinia Store：personaStore（全局共享）

```typescript
// stores/personaStore.ts

import { defineStore } from 'pinia'

interface PersonaState {
  /** 人设配置原始数据 | 📥存储 溯源：persona_config */
  config: PersonaConfig | null
  /** 缓存时间戳 */
  cachedAt: number | null
  /** 是否正在加载 */
  loading: boolean
}

const CACHE_TTL = 300000 // 5分钟

export const usePersonaStore = defineStore('persona', {
  state: (): PersonaState => ({
    config: null,
    cachedAt: null,
    loading: false,
  }),

  getters: {
    /** 缓存是否有效 | 溯源：数据流-人设配置全局缓存策略 */
    isCacheValid: (state): boolean => {
      if (!state.cachedAt) return false
      return Date.now() - state.cachedAt < CACHE_TTL
    },

    /** 开场白列表 | 📊计算 溯源：persona_config.catchphrases type="开场" */
    openingPhrases: (state): string[] =>
      state.config?.catchphrases
        ?.filter(c => c.type === '开场')
        .map(c => c.content) ?? [],

    /** 转折语列表 | 📊计算 溯源：persona_config.catchphrases type="转折" */
    transitionPhrases: (state): string[] =>
      state.config?.catchphrases
        ?.filter(c => c.type === '转折')
        .map(c => c.content) ?? [],

    /** 揭晓语列表 | 📊计算 溯源：persona_config.catchphrases type="揭晓" */
    revealPhrases: (state): string[] =>
      state.config?.catchphrases
        ?.filter(c => c.type === '揭晓')
        .map(c => c.content) ?? [],

    /** 结束语列表 | 📊计算 溯源：persona_config.catchphrases type="结束" */
    closingPhrases: (state): string[] =>
      state.config?.catchphrases
        ?.filter(c => c.type === '结束')
        .map(c => c.content) ?? [],

    /** 合并的样稿文件列表 | 📊计算 溯源：sample_texts + sample_audios */
    allSampleFiles: (state): SampleFile[] => {
      const texts = state.config?.sample_texts ?? []
      const audios = state.config?.sample_audios ?? []
      return [
        ...texts.map(f => ({ ...f, type: 'text' as const })),
        ...audios.map(f => ({ ...f, type: 'audio' as const })),
      ]
    },
  },

  actions: {
    /** 加载人设配置 | 溯源：数据流-GET /api/v1/persona */
    async loadConfig(): Promise<void> {
      if (this.config && this.isCacheValid) return
      this.loading = true
      try {
        this.config = await fetchPersonaConfig()
        this.cachedAt = Date.now()
      } finally {
        this.loading = false
      }
    },

    /** 获取人设配置（带缓存，供其他页面调用） */
    async getConfig(): Promise<PersonaConfig> {
      if (this.config && this.isCacheValid) {
        return this.config
      }
      await this.loadConfig()
      return this.config!
    },

    /** 保存配置 | 溯源：数据流-PUT /api/v1/persona */
    async saveConfig(params: SavePersonaConfigParams): Promise<void> {
      const result = await savePersonaConfig(params)
      // 保存成功后刷新缓存
      await this.loadConfig()
      // 触发全局缓存失效通知
      this.clearCache()
    },

    /** 上传样稿 | 溯源：数据流-POST /api/v1/persona/samples */
    async uploadSample(params: UploadSampleParams): Promise<void> {
      const result = await uploadSampleFile(params)
      // 刷新配置以更新样稿列表
      await this.loadConfig()
    },

    /** 删除样稿 | 溯源：数据流-DELETE /api/v1/persona/samples/:id */
    async deleteSample(sampleId: number): Promise<void> {
      await deleteSampleFile(sampleId)
      await this.loadConfig()
    },

    /** 清除缓存 | 溯源：数据流-人设配置全局缓存策略 */
    clearCache(): void {
      this.config = null
      this.cachedAt = null
      // 通知其他页面清除缓存（事件总线）
      window.dispatchEvent(new CustomEvent('persona-config-updated'))
    },
  },
})
```

### 5.3 表单脏检测（isDirty）

```typescript
// composables/usePersonaForm.ts

/** 判断表单是否有未保存修改 | 📊计算 溯源：数据流-表单是否修改 */
const isDirty = computed((): boolean => {
  if (!originalConfig.value) return Object.values(formBasic).some(v => v !== '' && v !== null)
  return (
    formBasic.display_name !== originalConfig.value.display_name ||
    formBasic.age !== originalConfig.value.age ||
    formBasic.career_background !== originalConfig.value.career_background ||
    formBasic.years_of_experience !== originalConfig.value.years_of_experience ||
    formBasic.core_experience !== originalConfig.value.core_experience ||
    formPosition.one_line_positioning !== originalConfig.value.one_line_positioning ||
    formPosition.core_philosophy !== originalConfig.value.core_philosophy ||
    JSON.stringify(formPosition.unique_selling_points) !== JSON.stringify(originalConfig.value.unique_selling_point) ||
    JSON.stringify(selectedStyles.value) !== JSON.stringify(originalConfig.value.language_styles) ||
    styleDescription.value !== originalConfig.value.style_description ||
    JSON.stringify(catchphrases.value) !== JSON.stringify(mapCatchphrasesFromConfig(originalConfig.value.catchphrases)) ||
    narrativePov.value !== originalConfig.value.narrative_viewpoint ||
    JSON.stringify(sampleFiles.value) !== JSON.stringify([
      ...originalConfig.value.sample_texts.map(f => ({ ...f, type: 'text' })),
      ...originalConfig.value.sample_audios.map(f => ({ ...f, type: 'audio' })),
    ])
  )
```
