## 4. 数据库表结构

### 4.1 内容资产表 (content_assets)

```sql
CREATE TABLE content_assets (
  id                    SERIAL PRIMARY KEY,
  title                 VARCHAR(500) NOT NULL,             -- 资产标题
  type                  VARCHAR(20) NOT NULL,               -- 类型: script/video/image_text
  status                VARCHAR(20) NOT NULL DEFAULT 'published', -- 状态: draft/published/archived
  video_product_id      INTEGER,                            -- 外键: 视频成品.id (可空, 脚本类型无关联)
  publish_platforms     VARCHAR(20)[] NOT NULL DEFAULT '{}', -- 发布平台数组
  published_at          TIMESTAMP,                          -- 发布时间
  archived_at           TIMESTAMP,                          -- 归档时间
  core_metrics          JSONB DEFAULT '{}',                 -- 核心数据指标
  /*
    core_metrics 结构:
    {
      "play_count": 12345,
      "like_count": 100,
      "comment_count": 50,
      "collect_count": 30,
      "completion_rate": 45.2,
      "private_message_count": 10,
      "interaction_rate": 1.5
    }
  */
  element_highlights    TEXT[] DEFAULT '{}',                 -- 元素亮点数组
  custom_tags           VARCHAR(50)[] DEFAULT '{}',          -- 手动标签数组
  reusable_materials    JSONB DEFAULT '[]',                  -- 可复用素材
  /*
    reusable_materials 结构:
    [
      {
        "material_id": "mat_001",
        "name": "背景音乐A",
        "thumbnail": "https://...",
        "copyright_status": "valid",   // valid / expiring_soon / expired
        "expires_at": "2026-06-30"
      }
    ]
  */
  cover_url             VARCHAR(500),                       -- 缩略图/封面URL (脚本类型为空)
  created_at            TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at            TIMESTAMP NOT NULL DEFAULT NOW(),

  CONSTRAINT fk_content_assets_video_product
    FOREIGN KEY (video_product_id) REFERENCES video_products(id),
  CONSTRAINT chk_content_assets_type
    CHECK (type IN ('script', 'video', 'image_text')),
  CONSTRAINT chk_content_assets_status
    CHECK (status IN ('draft', 'published', 'archived'))
);

-- 索引: 多维度筛选 (status + type + created_at 复合索引)
CREATE INDEX idx_content_assets_filter
  ON content_assets (status, type, created_at);

-- 索引: 平台数组筛选 (GIN索引支持 @> 操作)
CREATE INDEX idx_content_assets_platforms
  ON content_assets USING GIN (publish_platforms);

-- 索引: 核心数据指标 (GIN索引支持 JSONB 查询)
CREATE INDEX idx_content_assets_core_metrics
  ON content_assets USING GIN (core_metrics);

-- 索引: 外键索引(安全底线)
CREATE INDEX idx_content_assets_video_product_id
  ON content_assets (video_product_id);

-- 索引: 全文搜索 (标题 + 标签)
CREATE INDEX idx_content_assets_search
  ON content_assets USING GIN (to_tsvector('simple', coalesce(title, '') || ' ' || array_to_string(custom_tags, ' ')));

-- 索引: 自动归档查询 (定时任务使用)
CREATE INDEX idx_content_assets_auto_archive
  ON content_assets (status, published_at)
  WHERE status = 'published';
```

### 4.3 表现标签计算逻辑

```typescript
/** 性能标签阈值 (规则9) */
const PERFORMANCE_THRESHOLDS = {
  high_completion: { field: 'completion_rate', operator: '>=', value: 40 },  // 高完播 >= 40%
  high_interaction: { field: 'interaction_rate', operator: '>=', value: 5 }, // 高互动 >= 5%
  high_conversion: { field: 'private_message_count', operator: '>=', value: 10 }, // 高转化 >= 10 (注: 规则写>=10%但私信数为整数)
} as const

// 爆款结构: play_count >= AVG(all play_count) * 2
// 互动率公式: (like_count + comment_count + collect_count) / play_count * 100 (规则5)

/** 计算单个资产的表现标签 */
function calcPerformanceTags(metrics: CoreMetrics, avgPlayCount: number): PerformanceTag[] {
  const tags: PerformanceTag[] = []
  if (metrics.completion_rate >= 40) tags.push('high_completion')
  const engagementRate = metrics.play_count > 0
    ? (metrics.like_count + metrics.comment_count + metrics.collect_count) / metrics.play_count * 100
    : 0
  if (engagementRate >= 5) tags.push('high_interaction')
  if (metrics.private_message_count >= 10) tags.push('high_conversion')
  if (metrics.play_count >= avgPlayCount * 2) tags.push('viral_structure')
  return tags
}
```

/** 互动率计算公式 (规则5) */
function calcEngagementRate(metrics: CoreMetrics): string {
  if (metrics.play_count === 0) return '--'
  const rate = (metrics.like_count + metrics.comment_count + metrics.collect_count)
    / metrics.play_count * 100
  return rate.toFixed(1)
}

/** 格式化大数字: >=10000 显示 "X.X万" */
function formatLargeNumber(value: number): string {
  if (value >= 10000) {
    return (value / 10000).toFixed(1) + '万'
  }
  return value.toLocaleString()
}

/** 判断版权状态 */
function calcCopyrightStatus(expiresAt: string | null): CopyrightStatus {
  if (!expiresAt) return 'valid'
  const expires = new Date(expiresAt)
  const now = new Date()
  const weekLater = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000)
  if (expires < now) return 'expired'
  if (expires < weekLater) return 'expiring_soon'
  return 'valid'
}

/** 判断是否有版权过期素材 */
function hasExpiredMaterial(materials: MaterialItem[]): boolean {
  return materials.some(m => m.copyright_status === 'expired')
}

}

## 附录D：分页参数规范

所有列表 API 统一使用 `page` / `page_size` 参数 (规则11):
- `page`: 从1开始的页码, 默认1
- `page_size`: 每页条数, 默认20

- 资产库: 滚动到底自动加载更多 (累计追加 items)
