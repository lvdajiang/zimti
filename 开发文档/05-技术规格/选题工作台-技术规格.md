### 5.1 Pinia Store：topicStore

```typescript
// stores/topicStore.ts

import { defineStore } from 'pinia'

interface TopicState {
  /** 对标视频列表 | 📥存储 溯源：benchmark_videos */
  benchmarkVideos: BenchmarkVideoItem[]

  /** 热点推荐列表 | 📥存储 溯源：hot_topics */
  hotTopics: HotspotRecommendItem[]

  /** 已选对标视频ID数组 | 📊计算 溯源：前端状态 */
  selectedVideoIds: number[]

  /** 已选热点ID数组 | 📊计算 溯源：前端状态 */
  selectedHotspotIds: number[]

  /** 选题方案列表 | 📥存储 溯源：topic_proposals */
  proposals: TopicProposalItem[]

  /** 已选用选题ID | 📊计算 溯源：前端状态 */
  selectedProposalId: number | null

  /** AI生成状态 | 📊计算 溯源：前端状态 */
  generateStatus: 'idle' | 'generating' | 'completed' | 'failed'

  /** AI生成任务ID | 📊计算 溯源：API返回 */
  generateTaskId: string | null

  /** 脚本骨架数据 | 📥存储 溯源：选中proposal的子集 */
  scriptSkeleton: {
    topicTitle: string
    hook: string
    contentSkeleton: ContentSkeleton
    visualHighlights: string[]
    hotspotSuggestion: string
  } | null

  /** 融合模式标记 | 📊计算 溯源：前端状态 */
  isMerging: boolean

  /** 素材是否已变更（提示重新生成） | 📊计算 溯源：前端状态 */
  materialChanged: boolean
}

export const useTopicStore = defineStore('topic', {
  state: (): TopicState => ({
    benchmarkVideos: [],
    hotTopics: [],
    selectedVideoIds: [],
    selectedHotspotIds: [],
    proposals: [],
    selectedProposalId: null,
    generateStatus: 'idle',
    generateTaskId: null,
    scriptSkeleton: null,
    isMerging: false,
    materialChanged: false,
  }),

  getters: {
    /** 已选对标视频数量 */
    selectedVideoCount: (state): number => state.selectedVideoIds.length,

    /** 已选热点数量 */
    selectedHotspotCount: (state): number => state.selectedHotspotIds.length,

    /** 是否有选中的素材 */
    hasSelectedMaterial: (state): boolean =>
      state.selectedVideoIds.length > 0 || state.selectedHotspotIds.length > 0,

    /** 已选用选题数量（用于融合按钮状态） | 📊计算 溯源：selected_count */
    selectedProposalCount(): number {
      return this.proposals.filter(p => p.status === 'selected').length
    },

    /** 是否可融合（需 >= 2个已选用） | 📊计算 溯源：selected_count */
    canMerge(): boolean {
      return this.selectedProposalCount >= 2 && !this.isMerging
    },

    /** 是否可编辑脚本 | 📊计算 溯源：has_selection */
    canEditScript: (state): boolean => state.selectedProposalId !== null,

    /** 当前选中的选题详情 */
    selectedProposal: (state): TopicProposalItem | undefined =>
      state.proposals.find(p => p.id === state.selectedProposalId),
  },

  actions: {
    /** 切换对标视频选中状态 */
    toggleVideoSelection(videoId: number): void {
      const index = this.selectedVideoIds.indexOf(videoId)
      if (index >= 0) {
        this.selectedVideoIds.splice(index, 1)
      } else if (this.selectedVideoIds.length < 5) {
        this.selectedVideoIds.push(videoId)
      }
      if (this.proposals.length > 0) {
        this.materialChanged = true
      }
    },

    /** 切换热点选中状态 */
    toggleHotspotSelection(hotspotId: number): void {
      const index = this.selectedHotspotIds.indexOf(hotspotId)
      if (index >= 0) {
        this.selectedHotspotIds.splice(index, 1)
      } else {
        this.selectedHotspotIds.push(hotspotId)
      }
      if (this.proposals.length > 0) {
        this.materialChanged = true
      }
    },

    /** 选用选题（单选切换） */
    selectProposal(proposalId: number): void {
      if (this.selectedProposalId === proposalId) {
        this.selectedProposalId = null
        this.scriptSkeleton = null
      } else {
        this.selectedProposalId = proposalId
        const proposal = this.proposals.find(p => p.id === proposalId)
        if (proposal) {
          this.scriptSkeleton = {
            topicTitle: proposal.title,
            hook: proposal.core_hook,
            contentSkeleton: proposal.content_skeleton,
            visualHighlights: proposal.visual_highlights,
            hotspotSuggestion: '', // 需从 hot_topic_connections 关联查询
          }
        }
      }
    },

    /** 设置AI生成状态 */
    setGenerateStatus(status: 'idle' | 'generating' | 'completed' | 'failed'): void {
      this.generateStatus = status
    },

    /** 设置生成结果 */
    setProposals(proposals: TopicProposalItem[]): void {
      this.proposals = proposals
      this.selectedProposalId = null
      this.scriptSkeleton = null
      this.materialChanged = false
    },

    /** 添加融合结果 */
    addMergedProposal(proposal: TopicProposalItem): void {
      // 将原已选用方案标记为 merged
      this.proposals.forEach(p => {
        if (p.status === 'selected') {
          p.status = 'merged'
        }
      })
      this.proposals.push(proposal)
      this.selectedProposalId = proposal.id
      this.isMerging = false
    },

    /** 清除生成结果（换一批前） */
    clearProposals(): void {
      this.proposals = []
      this.selectedProposalId = null
      this.scriptSkeleton = null
    },
  },
})
```

// stores/personaStore.ts

interface PersonaState {
  /** 人设配置数据 | 🔗调用 溯源：persona_config */
  config: PersonaConfigSnippet | null
  /** 缓存时间戳 */
  cachedAt: number | null
  /** 缓存TTL（5分钟 = 300000ms） */
  readonly CACHE_TTL: 300000
}

export const usePersonaStore = defineStore('persona', {
  state: (): PersonaState => ({
    config: null,
    cachedAt: null,
  }),

  getters: {
    /** 缓存是否有效 */
    isCacheValid: (state): boolean => {
      if (!state.cachedAt) return false
      return Date.now() - state.cachedAt < 300000
    },
  },

  actions: {
    /** 获取人设配置（带缓存） */
    async getConfig(): Promise<PersonaConfigSnippet> {
      if (this.config && this.isCacheValid) {
        return this.config
      }
      const data = await fetchPersonaConfig()
      this.config = data
      this.cachedAt = Date.now()
      return data
    },

    /** 清除缓存（人设保存后调用） */
    clearCache(): void {
      this.config = null
      this.cachedAt = null
    },
  },
})

/** 解析数组形式的路由参数（支持逗号分隔或数组） */
function parseArrayParam(param: string | string[] | undefined): string[] {
  if (!param) return []
  if (Array.isArray(param)) return param
  return param.split(',').filter(Boolean)
}
