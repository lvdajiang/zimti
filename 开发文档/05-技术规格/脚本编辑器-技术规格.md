## 4. 数据库表结构

### 4.1 scripts（脚本表）

```sql
CREATE TABLE scripts (
  id                  SERIAL PRIMARY KEY,
  topic_id            INTEGER NOT NULL REFERENCES topic_proposals(id) ON DELETE CASCADE,
  task_id             INTEGER REFERENCES tasks(id) ON DELETE SET NULL,
  full_text           TEXT,                                          -- 📥存储 口播文案全文
  video_type          VARCHAR(20) NOT NULL DEFAULT 'knowledge',     -- 📥存储 视频类型：knowledge/emotional/viral
  oral_ratio          DECIMAL(5,2),                                  -- 📥存储 口播占比
  estimated_duration  INTEGER,                                       -- 📥存储 预估时长（秒）
  ai_check_result     JSONB,                                         -- 📥存储 去AI味检查结果
  status              VARCHAR(20) NOT NULL DEFAULT 'draft',         -- 📥存储 脚本状态
  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_scripts_topic_id ON scripts(topic_id);
CREATE INDEX idx_scripts_task_id ON scripts(task_id);
CREATE INDEX idx_scripts_status ON scripts(status);

-- updated_at 自动更新触发器
CREATE TRIGGER trigger_scripts_updated_at
  BEFORE UPDATE ON scripts
  FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
```

**字段溯源对照**：

| 字段名 | 数据流来源 | 页面描述元素 |
|--------|-----------|-------------|
| full_text | scripts.full_text | 区块B-左/#2 script_text |

### 5.1 Pinia Store：scriptStore

```typescript
// stores/scriptStore.ts

import { defineStore } from 'pinia'

interface ScriptState {
  /** 脚本基本信息 | 📥存储 溯源：scripts */
  script: ScriptData | null

  /** 分镜段落数组 | 📥存储 溯源：storyboard_segments */
  segments: StoryboardSegment[]

  /** 口播文案文本 | 📥存储 溯源：scripts.full_text */
  fullText: string

  /** 视频类型 | 📥存储 溯源：scripts.video_type */
  videoType: 'knowledge' | 'emotional' | 'viral'

  /** 口播占比 | 📊计算 溯源：voice_ratio_pct */
  voiceRatio: number

  /** 去AI味检查结果 | 🔗调用 溯源：scripts.ai_check_result */
  aiCheckResult: AiCheckIssue[] | null

  /** 去AI味面板可见 | 📊计算 溯源：前端状态 */
  aiCheckPanelVisible: boolean

  /** 导演脚本生成状态 | 📊计算 溯源：前端状态 */
  generateStatus: 'idle' | 'generating' | 'completed' | 'failed'

  /** 导演脚本生成任务ID | 📊计算 溯源：API返回 */
  generateTaskId: string | null

  /** 口播音频生成状态 | 📊计算 溯源：前端状态 */
  audioGenStatus: 'idle' | 'generating' | 'completed' | 'failed'

  /** 口播音频生成任务ID | 📊计算 溯源：API返回 */
  audioGenTaskId: string | null

  /** 素材需求列表 | 📥存储 溯源：storyboard_segments.required_materials */
  materials: MaterialListItem[]

  /** 素材面板展开状态 | 📊计算 溯源：前端状态 */
  materialPanelExpanded: boolean

  /** 选中分镜段落ID | 📊计算 溯源：前端状态 */
  activeSegmentId: number | null

  /** 编辑中段落ID | 📊计算 溯源：前端状态 */
  editingSegmentId: number | null

  /** 脚本ID（路由参数） | 📝用户输入 溯源：路由参数 */
  scriptId: number | null

  /** 选题ID（路由参数） | 📝用户输入 溯源：路由参数 */
  topicId: number | null
}

export const useScriptStore = defineStore('script', {
  state: (): ScriptState => ({
    script: null,
    segments: [],
    fullText: '',
    videoType: 'knowledge',
    voiceRatio: 0,
    aiCheckResult: null,
    aiCheckPanelVisible: false,
    generateStatus: 'idle',
    generateTaskId: null,
    audioGenStatus: 'idle',
    audioGenTaskId: null,
    materials: [],
    materialPanelExpanded: false,
    activeSegmentId: null,
    editingSegmentId: null,
    scriptId: null,
    topicId: null,
  }),

  getters: {
    /** 预估时长 | 📊计算 溯源：约3字/秒 */
    estimatedDuration(state): string {
      const charCount = state.fullText.length
      const seconds = Math.ceil(charCount / 3)
      const minutes = Math.floor(seconds / 60)
      const remainingSeconds = seconds % 60
      return `${minutes}:${String(remainingSeconds).padStart(2, '0')}`
    },

    /** 口播占比百分比 | 📊计算 溯源：口播文本占总文本比例 */
    voiceRatioPct(state): number {
      if (!state.fullText) return 0
      const totalOralChars = state.segments
        .reduce((sum, seg) => sum + (seg.oral_text?.length ?? 0), 0)
      return Math.round((totalOralChars / state.fullText.length) * 100)
    },

    /** 口播占比是否达标 | 📊计算 溯源：各类型阈值 */
    isVoiceRatioValid(state): boolean {
      const pct = this.voiceRatioPct
      switch (state.videoType) {
        case 'knowledge': return pct >= 70
        case 'emotional': return pct >= 50 && pct <= 70
        case 'viral': return pct >= 30 && pct <= 50
        default: return false
      }
    },

    /** 是否有严重AI味问题 | 📊计算 溯源：check_error_count */
    hasSevereAiIssue(state): boolean {
      return state.aiCheckResult?.some(i => i.severity === 'error') ?? false
    },

    /** 脚本是否完整 | 📊计算 溯源：script_complete */
    scriptComplete(state): boolean {
      const hasText = state.fullText.trim().length > 0
      const hasSegments = state.segments.length >= 1
      const noSevereIssue = !this.hasSevereAiIssue
      return hasText && hasSegments && noSevereIssue
    },

    /** 表单是否有未保存修改 | 📊计算 溯源：isDirty */
    isDirty(state): boolean {
      if (!state.script) return state.fullText.trim().length > 0
      return (
        state.fullText !== (state.script.full_text ?? '') ||
        state.videoType !== state.script.video_type ||
        state.segments.length > 0
      )
    },

    /** 去AI味检查统计 | 📊计算 溯源：check_result */
    aiCheckSummary(state): { passCount: number; warningCount: number; errorCount: number } {
      if (!state.aiCheckResult) return { passCount: 0, warningCount: 0, errorCount: 0 }
      return {
        passCount: 0, // 通过数不在问题列表中
        warningCount: state.aiCheckResult.filter(i => i.severity === 'warning').length,
        errorCount: state.aiCheckResult.filter(i => i.severity === 'error').length,
      }
    },

    /** 素材需求数量 | 📊计算 溯源：material_count */
    materialCount(state): number {
      return state.materials.length
    },
  },

  actions: {
    /** 初始化脚本数据 */
    initFromRoute(params: { script_id?: number; topic_id?: number }): void {
      this.scriptId = params.script_id ?? null
      this.topicId = params.topic_id ?? null
    },

    /** 加载脚本数据 | 溯源：GET /api/v1/scripts/:id */
    async loadScript(): Promise<void> {
      if (!this.scriptId) return
      const data = await fetchScript(this.scriptId)
      this.script = data
      this.fullText = data.full_text ?? ''
      this.videoType = data.video_type
      this.voiceRatio = data.oral_ratio
      this.aiCheckResult = data.ai_check_result
    },

    /** 加载分镜段落 | 溯源：GET /api/v1/scripts/:id/segments */
    async loadSegments(): Promise<void> {
      if (!this.scriptId) return
      this.segments = await fetchSegments(this.scriptId)
    },

    /** 加载素材需求 | 溯源：GET /api/v1/scripts/:id/materials */
    async loadMaterials(): Promise<void> {
      if (!this.scriptId) return
      this.materials = await fetchMaterials(this.scriptId)
    },

    /** 自动保存（静默） | 溯源：30秒间隔 */
    async autoSave(): Promise<void> {
      if (!this.scriptId || !this.isDirty) return
      try {
        await saveScript(this.scriptId, {
          full_text: this.fullText,
          video_type: this.videoType,
          oral_ratio: this.voiceRatioPct,
        })
      } catch {
        // 静默失败，不干扰用户
      }
    },

    /** 保存分镜段落 | 溯源：PUT /api/v1/scripts/:id/segments/{segment_id} */
    async saveSegmentData(segmentId: number, data: Partial<SegmentUpdateData>): Promise<void> {
      if (!this.scriptId) return
      await saveSegment(this.scriptId, segmentId, data as SaveSegmentParams)
      // 更新本地状态
      const idx = this.segments.findIndex(s => s.id === segmentId)
      if (idx >= 0) {
        Object.assign(this.segments[idx], data)
      }
    },

    /** 添加分镜段落 | 溯源：POST /api/v1/scripts/:id/segments */
    async addSegment(): Promise<void> {
      if (!this.scriptId) return
      const newIndex = this.segments.length
      const result = await addSegment(this.scriptId, {
        segment_index: newIndex,
        time_start: '00:00',
        time_end: '00:05',
        purpose: '',
        visual_instruction: '',
        audio_instruction: '',
        enhancement_instruction: '',
        oral_text: '',
        required_materials: [],
      })
      this.segments.push({
        id: result.id,
        script_id: this.scriptId,
        segment_index: newIndex,
        time_start: '00:00',
        time_end: '00:05',
        purpose: '',
        visual_instruction: '',
        audio_instruction: '',
        enhancement_instruction: '',
        oral_text: '',
        required_materials: [],
        created_at: new Date().toISOString(),
      })
    },

    /** 删除分镜段落 | 溯源：DELETE /api/v1/scripts/:id/segments/{segment_id} */
    async removeSegment(segmentId: number): Promise<void> {
      if (!this.scriptId) return
      await deleteSegment(this.scriptId, segmentId)
      this.segments = this.segments.filter(s => s.id !== segmentId)
      // 重排索引
      this.segments.forEach((s, i) => { s.segment_index = i })
    },

    /** 重新排序 | 溯源：PATCH /api/v1/scripts/:id/segments/reorder */
    async reorderSegments(newOrder: StoryboardSegment[]): Promise<void> {
      if (!this.scriptId) return
      this.segments = newOrder.map((s, i) => ({ ...s, segment_index: i }))
      await reorderSegments(this.scriptId, {
        segment_orders: this.segments.map(s => ({ id: s.id, segment_index: s.segment_index })),
      })
    },

    /** 应用AI味修改建议 | 溯源：PATCH /api/v1/scripts/:id/apply-suggestion */
    async applySuggestion(issueIndex: number): Promise<void> {
      if (!this.scriptId || !this.aiCheckResult) return
      const issue = this.aiCheckResult[issueIndex]
      const result = await applySuggestion(this.scriptId, {
        original_text: issue.originalText,
        suggestion: issue.suggestion,
        start: issue.start,
        end: issue.end,
      })
      this.fullText = result.data.full_text
      issue.applied = true
    },

    /** 设置AI检查结果 */
    setAiCheckResult(result: AiCheckIssue[]): void {
      this.aiCheckResult = result
      this.aiCheckPanelVisible = true
    },
  },
})
```

// composables/useAutoSave.ts

/** 自动保存 Hook | 溯源：30秒间隔 */
function useAutoSave(store: ReturnType<typeof useScriptStore>) {
  const isDirty = computed(() => store.isDirty)
  let timer: ReturnType<typeof setInterval> | null = null

  onMounted(() => {
    timer = setInterval(() => {
      if (isDirty.value) {
        store.autoSave()
      }
    }, 30000) // 30秒
  })

  onUnmounted(() => {
    if (timer) clearInterval(timer)
  })
}
