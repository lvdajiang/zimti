## 4. 数据库表结构

### 4.1 video_products 表（视频成品）

```sql
CREATE TABLE video_products (
  id                      UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  script_id               INTEGER NOT NULL REFERENCES scripts(id),     -- 关联分镜脚本（scripts.id 为 SERIAL）
  task_id                 VARCHAR(100),                                -- 渲染任务ID（Remotion Lambda task_id）
  title                   VARCHAR(255) NOT NULL,                       -- 视频标题
  resolution              VARCHAR(10) NOT NULL DEFAULT '1080p',        -- 分辨率：1080p/720p/480p
  duration                DECIMAL(10,2) NOT NULL DEFAULT 0,            -- 视频时长（秒）
  video_url               TEXT,                                        -- 渲染完成后的视频文件URL
  file_size               BIGINT DEFAULT 0,                            -- 文件大小（字节）
  status                  VARCHAR(20) NOT NULL DEFAULT 'draft',        -- 状态：draft/rendering/completed/failed
  render_config           JSONB DEFAULT '{}'::jsonb,                   -- 渲染配置（字幕/音量/转场等）
  platform                VARCHAR(20) DEFAULT 'main',                 -- 平台标识：main/douyin/weixin/image_text
  parent_video_product_id UUID REFERENCES video_products(id),          -- 多版本关联的父版本ID
  created_at              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at              TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  CONSTRAINT chk_vp_status CHECK (status IN ('draft', 'rendering', 'completed', 'failed')),
  CONSTRAINT chk_vp_resolution CHECK (resolution IN ('1080p', '720p', '480p')),
  CONSTRAINT chk_vp_platform CHECK (platform IN ('main', 'douyin', 'weixin', 'image_text'))
);

-- 索引
CREATE INDEX idx_vp_script_id ON video_products(script_id);
CREATE INDEX idx_vp_status ON video_products(status);
CREATE INDEX idx_vp_parent_id ON video_products(parent_video_product_id);
CREATE INDEX idx_vp_task_id ON video_products(task_id);
```

```typescript
import { defineStore } from 'pinia'

export const useVideoPreviewStore = defineStore('videoPreview', () => {
  // ===== 路由参数 =====
  /** 脚本ID，溯源：数据流-前端状态 script_id */
  const scriptId = ref<string>('')

  // ===== 核心数据 =====
  /** 脚本信息，溯源：数据流-前端状态 script */
  const script = ref<Script | null>(null)

  /** 分镜段落列表，溯源：数据流-前端状态 segments[] */
  const segments = ref<Segment[]>([])

  /** 素材映射表（material_id -> Material），溯源：数据流-前端状态 materials_map */
  const materialsMap = ref<Map<string, Material>>(new Map())

  /** 口播音频列表，溯源：数据流-前端状态 oral_audios[] */
  const oralAudios = ref<OralAudio[]>([])

  // ===== 时间轴状态 =====
  /** 时间轴片段列表（从segments构建），溯源：数据流-前端状态 timeline_segments[] */
  const timelineSegments = ref<TimelineSegment[]>([])

  /** 选中的片段ID，溯源：数据流-前端状态 selected_segment_id */
  const selectedSegmentId = ref<string | null>(null)

  /** 播放头位置（秒），溯源：数据流-前端状态 playhead_position */
  const playheadPosition = ref(0)

  /** 是否正在播放，溯源：数据流-前端状态 is_playing */
  const isPlaying = ref(false)

  /** 时间轴缩放比例（50%-200%） */
  const zoomLevel = ref(100)

  // ===== 渲染配置 =====
  /** 分辨率，溯源：数据流-前端状态 resolution */
  const resolution = ref<Resolution>('1080p')

  /** 右侧面板标签，溯源：数据流-前端状态 panel_tab */
  const panelTab = ref<PanelTab>('素材')

  /** 渲染配置，溯源：数据流-前端状态 render_config */
  const renderConfig = ref<RenderConfig>({
    subtitle_size: 16,
    subtitle_color: '#FFFFFF',
    subtitle_position: '底部居中',
    bgm_volume: 80,
    voice_volume: 100,
    sfx_volume: 100
  })

  // ===== Remotion相关 =====
  /** 预览URL，溯源：数据流-前端状态 preview_url */
  const previewUrl = ref<string | null>(null)

  /** Remotion合成配置，溯源：数据流-前端状态 composition_config */
  const compositionConfig = ref<CompositionConfig | null>(null)

  // ===== 渲染状态 =====
  /** 视频成品ID，溯源：数据流-前端状态 video_product_id */
  const videoProductId = ref<string | null>(null)

  /** 渲染任务ID，溯源：数据流-前端状态 render_task_id */
  const renderTaskId = ref<string | null>(null)

  /** 渲染进度，溯源：数据流-前端状态 render_progress */
  const renderProgress = ref(0)

  /** 渲染状态 */
  const renderStatus = ref<RenderStatus>('idle')

  /** 渲染完成后视频URL */
  const renderVideoUrl = ref<string | null>(null)

  /** 渲染轮询定时器 */
  let renderPollTimer: ReturnType<typeof setTimeout> | null = null

  // ===== 弹窗状态 =====
  /** 渲染导出弹窗，溯源：数据流-前端状态 show_render_modal */
  const showRenderModal = ref(false)

  /** 多版本生成弹窗，溯源：数据流-前端状态 show_multi_version_modal */
  const showMultiVersionModal = ref(false)

  // ===== 多版本状态 =====
  /** 平台勾选状态 */
  const genDouyin = ref(true)
  const genXiaohongshu = ref(true)
  const genWeixin = ref(true)

  /** 各平台渲染进度，溯源：数据流-前端状态 platform_progress_map */
  const platformProgressMap = ref<Map<string, PlatformRenderStatus>>(new Map())

  // ===== 其他状态 =====
  /** 是否有未保存修改，溯源：数据流-前端状态 has_unsaved_changes */
  const hasUnsavedChanges = ref(false)

  /** 加载状态 */
  const loading = ref(false)

  // ===== 计算属性 =====
  /** 视频标题 */
  const videoTitle = computed(() => {
    if (!script.value) return ''
    return script.value.title || script.value.full_text?.slice(0, 30) || ''
  })

  /** 预估时长（格式化） */
  const estimatedDuration = computed(() => {
    const total = timelineSegments.value.reduce(
      (sum, s) => sum + (s.time_end - s.time_start), 0
    )
    const minutes = Math.floor(total / 60)
    const seconds = Math.floor(total % 60)
    return `${minutes}:${seconds.toString().padStart(2, '0')}`
  })

  /** 总时长（秒） */
  const totalDuration = computed(() => {
    return timelineSegments.value.reduce(
      (sum, s) => sum + (s.time_end - s.time_start), 0
    )
  })

  /** 当前选中片段 */
  const selectedSegment = computed(() => {
    if (!selectedSegmentId.value) return null
    return timelineSegments.value.find(s => s.id === selectedSegmentId.value) ?? null
  })

  /** 选中片段的素材列表 */
  const selectedSegmentMaterials = computed(() => {
    if (!selectedSegment.value) return []
    return selectedSegment.value.material_ids
      .map(id => materialsMap.value.get(id))
      .filter((m): m is Material => m !== undefined)
  })

  /** 主版本是否已完成 */
  const isMainVersionCompleted = computed(() => {
    return renderStatus.value === 'completed'
  })

  /** 预估文件大小 */
  const estimatedFileSize = computed(() => {
    const minutes = totalDuration.value / 60
    const rateMap: Record<Resolution, number> = {
      '1080p': 10, // 10MB/min
      '720p': 5,
      '480p': 2.5
    }
    const sizeMB = minutes * (rateMap[resolution.value] || 10)
    return `${sizeMB.toFixed(1)}MB`
  })

  // ===== Actions =====

  /** 初始化页面，溯源：数据流-状态流转"页面初始化状态流转" */
  async function initPage(id: string): Promise<void> {
    scriptId.value = id
    loading.value = true
    try {
      // 1. 加载分镜数据
      const storyboard = await videoPreviewService.getStoryboard(id)
      script.value = storyboard.script
      segments.value = storyboard.segments
      // 构建素材映射
      const map = new Map<string, Material>()
      storyboard.materials.forEach(m => map.set(m.id, m))
      materialsMap.value = map
      oralAudios.value = storyboard.oral_audios

      // 2. 构建时间轴片段
      buildTimelineSegments()

      // 3. 选中第一个片段
      if (timelineSegments.value.length > 0) {
        selectedSegmentId.value = timelineSegments.value[0].id
      }

      // 4. 获取Remotion合成配置
      const comp = await videoPreviewService.getComposition(id)
      compositionConfig.value = comp
    } finally {
      loading.value = false
    }
  }

  /** 从segments构建timelineSegments */
  function buildTimelineSegments(): void {
    let offset = 0
    timelineSegments.value = segments.value
      .sort((a, b) => a.segment_index - b.segment_index)
      .map(seg => {
        const ts: TimelineSegment = {
          id: seg.id,
          name: seg.purpose,
          duration: seg.time_end - seg.time_start,
          time_start: offset,
          time_end: offset + (seg.time_end - seg.time_start),
          sort_index: seg.segment_index,
          transition_type: seg.transition_type,
          thumbnail_url: null,
          material_ids: seg.required_materials
        }
        offset = ts.time_end
        return ts
      })
  }

  /** 播放/暂停 */
  function togglePlay(): void {
    isPlaying.value = !isPlaying.value
  }

  /** 拖拽排序片段，溯源：数据流-状态流转"片段编辑状态流转" */
  async function reorderSegments(newOrder: string[]): Promise<void> {
    // 乐观更新
    const oldSegments = [...timelineSegments.value]
    const reordered = newOrder.map(id => timelineSegments.value.find(s => s.id === id)!)
    timelineSegments.value = reordered
    hasUnsavedChanges.value = true

    try {
      await videoPreviewService.reorderSegments(scriptId.value, newOrder)
    } catch {
      // 回滚
      timelineSegments.value = oldSegments
    }
  }

  /** 调整片段时长 */
  async function resizeSegment(segmentId: string, newStart: number, newEnd: number): Promise<void> {
    const seg = timelineSegments.value.find(s => s.id === segmentId)
    if (!seg) return
    seg.time_start = newStart
    seg.time_end = newEnd
    seg.duration = newEnd - newStart
    hasUnsavedChanges.value = true

    // 防抖后调用API
    await videoPreviewService.updateSegment(scriptId.value, segmentId, {
      time_start: newStart,
      time_end: newEnd
    })
  }

  /** 修改转场效果 */
  async function updateTransition(segmentId: string, type: TransitionType): Promise<void> {
    const seg = timelineSegments.value.find(s => s.id === segmentId)
    if (!seg) return
    seg.transition_type = type
    hasUnsavedChanges.value = true

    await videoPreviewService.updateTransition(scriptId.value, segmentId, {
      transition_type: type
    })
  }

  /** 替换片段素材 */
  async function replaceMaterials(segmentId: string, materialIds: string[]): Promise<void> {
    const result = await videoPreviewService.replaceSegmentMaterials(
      scriptId.value, segmentId, { material_ids: materialIds }
    )
    // 更新本地数据
    const seg = timelineSegments.value.find(s => s.id === segmentId)
    if (seg) seg.material_ids = materialIds
    result.materials.forEach(m => materialsMap.value.set(m.id, m))
    hasUnsavedChanges.value = true
  }

  /** 更新渲染配置 */
  async function updateRenderConfig(config: Partial<RenderConfig>): Promise<void> {
    renderConfig.value = { ...renderConfig.value, ...config }
    hasUnsavedChanges.value = true
  }

  /** 发起渲染，溯源：数据流-状态流转"渲染导出状态流转" */
  async function startRender(): Promise<void> {
    const result = await videoPreviewService.startRender({
      script_id: scriptId.value,
      resolution: resolution.value,
      render_config: renderConfig.value
    })
    videoProductId.value = result.video_product_id
    renderTaskId.value = result.task_id
    renderStatus.value = 'rendering'
    renderProgress.value = 0

    // 开始轮询（3秒间隔）
    startRenderPolling()
  }

  /** 开始轮询渲染状态 */
  function startRenderPolling(): void {
    stopRenderPolling()
    renderPollTimer = setTimeout(async () => {
      if (!videoProductId.value) return
      try {
        const status = await videoPreviewService.getRenderStatus(videoProductId.value)
        renderProgress.value = status.progress
        if (status.status === 'rendering') {
          startRenderPolling() // 继续轮询
        } else if (status.status === 'completed') {
          renderStatus.value = 'completed'
          renderVideoUrl.value = status.video_url ?? null
        } else if (status.status === 'failed') {
          renderStatus.value = 'failed'
        }
      } catch {
        startRenderPolling() // 轮询出错，继续
      }
    }, 3000)
  }

  /** 停止轮询 */
  function stopRenderPolling(): void {
    if (renderPollTimer) {
      clearTimeout(renderPollTimer)
      renderPollTimer = null
    }
  }

  /** 取消渲染 */
  async function cancelRender(): Promise<void> {
    if (!videoProductId.value) return
    stopRenderPolling()
    await videoPreviewService.cancelRender(videoProductId.value)
    renderStatus.value = 'failed'
  }

  /** 批量生成多版本，溯源：数据流-状态流转"多版本生成状态流转" */
  async function deriveVersions(): Promise<void> {
    if (!videoProductId.value) return
    const platforms: Platform[] = []
    if (genDouyin.value) platforms.push('douyin')
    if (genXiaohongshu.value) platforms.push('xiaohongshu')
    if (genWeixin.value) platforms.push('weixin')

    const result = await videoPreviewService.deriveVersions(
      videoProductId.value, { platforms }
    )

    // 初始化各平台进度
    result.tasks.forEach(t => {
      platformProgressMap.value.set(t.platform, {
        platform: t.platform,
        status: 'rendering',
        progress: 0,
        video_url: null,
        error_message: null
      })
    })

    // 开始轮询多版本
    startDerivePolling()
  }

  /** 轮询多版本状态 */
  function startDerivePolling(): void {
    setTimeout(async () => {
      if (!videoProductId.value) return
      const status = await videoPreviewService.getDeriveStatus(videoProductId.value)
      status.tasks.forEach(t => {
        platformProgressMap.value.set(t.platform, {
          platform: t.platform,
          status: t.status,
          progress: t.progress,
          video_url: t.video_url ?? null,
          error_message: null
        })
      })
      // 检查是否全部完成
      const allDone = status.tasks.every(t => t.status === 'completed' || t.status === 'failed')
      if (!allDone) startDerivePolling()
    }, 3000)
  }

  /** 保存进度，溯源：数据流-状态流转"返回脚本编辑器" */
  async function saveProgress(): Promise<void> {
    await videoPreviewService.saveProgress(scriptId.value, {
      timeline_segments: timelineSegments.value,
      render_config: renderConfig.value
    })
    hasUnsavedChanges.value = false
  }

  return {
    // state
    scriptId, script, segments, materialsMap, oralAudios,
    timelineSegments, selectedSegmentId, playheadPosition, isPlaying, zoomLevel,
    resolution, panelTab, renderConfig,
    previewUrl, compositionConfig,
    videoProductId, renderTaskId, renderProgress, renderStatus, renderVideoUrl,
    showRenderModal, showMultiVersionModal,
    genDouyin, genXiaohongshu, genWeixin, platformProgressMap,
    hasUnsavedChanges, loading,
    // computed
    videoTitle, estimatedDuration, totalDuration, selectedSegment,
    selectedSegmentMaterials, isMainVersionCompleted, estimatedFileSize,
    // actions
    initPage, togglePlay, buildTimelineSegments,
    reorderSegments, resizeSegment, updateTransition,
    replaceMaterials, updateRenderConfig,
    startRender, startRenderPolling, stopRenderPolling, cancelRender,
    deriveVersions, saveProgress
  }
})
```

## 6. 路由配置

文件路径：`src/router/index.ts`

```typescript
{
  path: '/video-preview/:scriptId',
  name: 'VideoPreview',
  component: () => import('@/views/VideoPreviewPage.vue'),
  props: true,
  meta: {
    title: '视频预览与渲染',
    requiresAuth: true,
    keepAlive: false  // 预览页含Remotion播放器，不适合keepAlive
  }
}
```

### 路由参数

| 参数 | 类型 | 必填 | 说明 | 溯源 |
|------|------|------|------|------|
| scriptId | string (UUID) | 是 | 分镜脚本ID，从脚本编辑器"生成视频"按钮跳转时带入 | 数据流-跨页面参数 script_id |

### 跨页面导航

| 来源页面 | 导航方式 | 传递参数 | 接收方式 |
|---------|---------|---------|---------|
| 脚本编辑器 | `router.push({ name: 'VideoPreview', params: { scriptId } })` | script_id | `route.params.scriptId` |
| 返回脚本编辑器 | `router.push({ name: 'ScriptEditor', params: { scriptId } })` | script_id | — |
| 跳转素材库（替换素材） | `router.push({ path: '/materials', query: { segment: selectedSegmentId, select: 'true' } })` | segment_id | 素材库选择后通过事件/回调返回 |
| 渲染完成后跳转发布工作台 | `router.push({ name: 'PublishWorkbench', query: { video_product_id: videoProductId } })` | video_product_id | `route.query.video_product_id` |
| "在素材库中查看" | `router.push('/materials')` | 无 | — |
