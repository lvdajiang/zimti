## 4. 数据库表结构

### 4.1 数据快照表 (data_snapshots)

```sql
CREATE TABLE data_snapshots (
  id                    SERIAL PRIMARY KEY,
  publish_record_id     UUID NOT NULL,                     -- 外键: 平台发布记录.id（publish_records.id 为 UUID）
  play_count            INTEGER NOT NULL DEFAULT 0,         -- 播放量
  completion_rate       DECIMAL(5,2) NOT NULL DEFAULT 0,    -- 完播率(%), 保留2位小数
  three_second_bounce_rate DECIMAL(5,2) NOT NULL DEFAULT 0, -- 3秒跳出率(%)
  like_count            INTEGER NOT NULL DEFAULT 0,         -- 点赞数
  comment_count         INTEGER NOT NULL DEFAULT 0,         -- 评论数
  collect_count         INTEGER NOT NULL DEFAULT 0,         -- 收藏数
  private_message_count INTEGER NOT NULL DEFAULT 0,         -- 私信数
  snapshot_at           TIMESTAMP NOT NULL DEFAULT NOW(),   -- 快照时间
  created_at            TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at            TIMESTAMP NOT NULL DEFAULT NOW(),

  CONSTRAINT fk_data_snapshots_publish_record
    FOREIGN KEY (publish_record_id) REFERENCES publish_records(id)
);

-- 索引: 按发布记录+时间范围聚合查询
CREATE INDEX idx_data_snapshots_record_time
  ON data_snapshots (publish_record_id, snapshot_at);

-- 索引: 外键索引(安全底线)
CREATE INDEX idx_data_snapshots_publish_record_id
  ON data_snapshots (publish_record_id);
```

/** 互动率计算公式 (规则5) */
function calcEngagementRate(
  likeCount: number,
  commentCount: number,
  collectCount: number,
  playCount: number
): string {
  if (playCount === 0) return '--'
  const rate = (likeCount + commentCount + collectCount) / playCount * 100
  return rate.toFixed(1)
}

/** 格式化大数字: >=10000 显示 "X.X万" */
function formatLargeNumber(value: number): string {
  if (value >= 10000) {
    return (value / 10000).toFixed(1) + '万'
  }
  return value.toLocaleString()
}

/** 格式化趋势百分比 */
function formatTrend(current: number, previous: number): TrendValue {
  if (previous === 0) {
    return { direction: 'flat', percentage: 0, label: '--' }
  }
  const pct = ((current - previous) / previous) * 100
  const direction = pct > 0 ? 'up' : pct < 0 ? 'down' : 'flat'
  const arrow = direction === 'up' ? '↑' : direction === 'down' ? '↓' : '-'
  const label = `${arrow}${Math.abs(pct).toFixed(0)}% vs上周期`
  return { direction, percentage: pct, label }
}

/** AI分析轮询 (规则8: 3秒轮询, 30秒超时) */
async function pollAiAnalysis(
  taskId: string,
  onStatusChange: (status: AnalysisStatus) => void
): Promise<AiAnalysisResultResponse> {
  const MAX_ATTEMPTS = 10  // 10次 * 3秒 = 30秒
  const INTERVAL = 3000   // 3秒

  for (let i = 0; i < MAX_ATTEMPTS; i++) {
    await sleep(INTERVAL)
    const result = await getAiAnalysisResult(taskId)
    if (result.status === 'completed') return result
    if (result.status === 'failed') throw new Error('AI分析失败')
  }
  throw new Error('AI分析超时(30秒)')
}
```

    endDate: route.query.end_date as string || undefined,
    page: Number(route.query.page) || 1
  })
}
```

### 6.1 URL Query 参数

| 参数名 | 类型 | 默认值 | 说明 |
|--------|------|--------|------|
| time_range | string | this_week | 时间范围筛选 |
| start_date | string | - | 自定义开始日期(YYYY-MM-DD) |
| end_date | string | - | 自定义结束日期(YYYY-MM-DD) |
| page | number | 1 | 当前页码 |
