## 4. 数据库表结构

### 4.1 publish_records 表（平台发布记录）

```sql
CREATE TABLE publish_records (
  id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  video_product_id    UUID NOT NULL REFERENCES video_products(id),      -- 关联视频成品
  platform            VARCHAR(20) NOT NULL,                              -- 📥存储 平台：xiaohongshu/douyin/weixin
  title               VARCHAR(100) DEFAULT '',                           -- 📥存储 平台标题，最大30字
  description         TEXT DEFAULT '',                                   -- 📥存储 平台描述，最大500字
  tags                JSONB DEFAULT '[]'::jsonb,                         -- 📥存储 话题标签数组
  comment_guide       JSONB DEFAULT '[]'::jsonb,                        -- 📥存储 引导话术数组
  reply_templates     JSONB DEFAULT '[]'::jsonb,                        -- 📥存储 回复建议数组
  geo_content         JSONB DEFAULT '{"items":[]}'::jsonb,              -- 📥存储 GEO图文内容
  seo_score           INTEGER,                                           -- 📥存储 SEO评分（0-100）
  seo_checks          JSONB DEFAULT '[]'::jsonb,                        -- 📥存储 SEO检查结果
  publish_status      VARCHAR(20) NOT NULL DEFAULT 'unpublished',        -- 📥存储 状态：unpublished/pending/published/failed
  published_at        TIMESTAMPTZ,                                       -- 📥存储 发布时间
  conversion_type     VARCHAR(20) DEFAULT 'awareness',                   -- 📥存储 转化目标
  aigc_reminded       BOOLEAN NOT NULL DEFAULT false,                    -- 📥存储 AIGC标识确认状态
  created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),

  CONSTRAINT chk_pr_platform CHECK (platform IN ('xiaohongshu', 'douyin', 'weixin')),
  CONSTRAINT chk_pr_publish_status CHECK (publish_status IN ('unpublished', 'pending', 'published', 'failed')),
  CONSTRAINT chk_pr_conversion_type CHECK (conversion_type IN ('awareness', 'trust', 'conversion')),
  CONSTRAINT chk_pr_seo_score CHECK (seo_score IS NULL OR (seo_score >= 0 AND seo_score <= 100)),
  CONSTRAINT uq_pr_video_platform UNIQUE(video_product_id, platform)
);

-- 外键索引（安全底线要求）
CREATE INDEX idx_pr_video_product_id ON publish_records(video_product_id);
CREATE INDEX idx_pr_platform ON publish_records(platform);
CREATE INDEX idx_pr_publish_status ON publish_records(publish_status);
```

```typescript
  // ===== 核心数据 =====
  /** 视频信息 */
  const video = ref<VideoProductBrief | null>(null)

  /** 脚本信息 */
  const scriptInfo = ref<ScriptBrief | null>(null)

  /** 发布记录映射（按platform索引），溯源：数据流-前端状态 publish_records_map */
  const publishRecordsMap = ref<Map<Platform, PublishRecord>>(new Map())

  // ===== 平台与文案状态 =====
  /** 当前平台，溯源：数据流-前端状态 current_platform */
  const currentPlatform = ref<Platform>('xiaohongshu')

  /** 转化目标，溯源：数据流-前端状态 conversion_goal */
  const conversionGoal = ref<ConversionGoal>('awareness')

  /** AIGC确认状态，溯源：数据流-前端状态 aigc_confirmed */
  const aigcConfirmed = ref(false)

  /** 平台标题，溯源：数据流-前端状态 platform_title */
  const platformTitle = ref('')

  /** 平台描述，溯源：数据流-前端状态 platform_description */
  const platformDescription = ref('')

  /** 话题标签，溯源：数据流-前端状态 tags[] */
  const tags = ref<string[]>([])

  // ===== SEO状态 =====
  /** SEO评分，溯源：数据流-前端状态 seo_score */
  const seoScore = ref<number | null>(null)

  /** SEO检查项，溯源：数据流-前端状态 seo_checks[] */
  const seoChecks = ref<SeoCheck[]>([])

  /** SEO检查中 */
  const isSeoChecking = ref(false)

  // ===== 评论区状态 =====
  /** 评论标签，溯源：数据流-前端状态 comment_tab */
  const commentTab = ref<CommentTab>('引导话术')

  /** 引导话术列表，溯源：数据流-前端状态 comment_guide_items[] */
  const commentGuideItems = ref<CommentItem[]>([])

  /** 回复建议列表，溯源：数据流-前端状态 reply_template_items[] */
  const replyTemplateItems = ref<CommentItem[]>([])

  // ===== GEO状态 =====
  /** GEO图文列表，溯源：数据流-前端状态 geo_items[] */
  const geoItems = ref<GeoItem[]>([])

  /** GEO生成中 */
  const isGeneratingGeo = ref(false)

  // ===== 发布状态 =====
  /** 各平台发布状态列表，溯源：数据流-前端状态 publish_status[] */
  const platformStatuses = ref<PlatformPublishStatus[]>([])

  // ===== 加载状态 =====
  /** AI生成文案中 */
  const isGeneratingCopy = ref(false)

  /** 是否有未保存修改，溯源：数据流-前端状态 has_unsaved_changes */
  const hasUnsavedChanges = ref(false)

  /** 自动保存定时器 */
  let autoSaveTimer: ReturnType<typeof setTimeout> | null = null

  /** 加载状态 */
  const loading = ref(false)

  // ===== 计算属性 =====
  /** 当前平台发布记录ID */
  const currentRecordId = computed(() => {
    return publishRecordsMap.value.get(currentPlatform.value)?.id ?? null
  })

  /** 标题字数，溯源：数据流-显示字段 title_length */
  const titleLength = computed(() => platformTitle.value.length)

  /** 描述字数，溯源：数据流-显示字段 desc_length */
  const descLength = computed(() => platformDescription.value)

  /** 当前平台标签数量上限 */
  const tagLimit = computed(() => PLATFORM_TAG_LIMITS[currentPlatform.value])

  /** 是否达到标签上限 */
  const isTagLimitReached = computed(() => tags.value.length >= tagLimit.value)

  /** "标记已发布"按钮是否可用（SEO>=40） */
  const canPublish = computed(() => {
    if (seoScore.value === null) return false
    return seoScore.value >= 40
  })

  // ===== Actions =====

  /** 初始化页面，溯源：数据流-状态流转"页面初始化状态流转" */
  async function initPage(vpId: string): Promise<void> {
    videoProductId.value = vpId
    loading.value = true
    try {
      // 1. 加载发布工作台数据
      const workspace = await publishService.getPublishWorkspace(vpId)
      video.value = workspace.video
      scriptInfo.value = workspace.script

      // 2. 构建发布记录Map
      const map = new Map<Platform, PublishRecord>()
      workspace.publish_records.forEach(r => map.set(r.platform, r))
      publishRecordsMap.value = map

      // 3. 加载默认平台（小红书）文案
      loadPlatformData('xiaohongshu')

      // 4. 构建发布状态列表
      buildPlatformStatuses()

      // 5. 加载转化目标
      if (scriptInfo.value?.task_id) {
        try {
          const goalResult = await publishService.getConversionGoal(scriptInfo.value.task_id)
          conversionGoal.value = goalResult.conversion_type
        } catch {
          // 转化目标加载失败不影响主流程
        }
      }

      // 6. 如果有SEO评分，加载SEO检查项
      if (workspace.seo_score !== undefined) {
        seoScore.value = workspace.seo_score
      }
      if (workspace.seo_checks) {
        seoChecks.value = workspace.seo_checks
      }
    } finally {
      loading.value = false
    }
  }

  /** 加载指定平台数据，溯源：数据流-状态流转"平台切换状态流转" */
  function loadPlatformData(platform: Platform): void {
    const record = publishRecordsMap.value.get(platform)
    if (record) {
      platformTitle.value = record.title
      platformDescription.value = record.description
      tags.value = [...record.tags]
      commentGuideItems.value = record.comment_guide.map((text, i) => ({ id: `guide-${i}`, text }))
      replyTemplateItems.value = record.reply_templates.map((text, i) => ({ id: `reply-${i}`, text }))
      geoItems.value = record.geo_content?.items ?? []
      seoScore.value = record.seo_score
      aigcConfirmed.value = record.aigc_reminded
    } else {
      platformTitle.value = ''
      platformDescription.value = ''
      tags.value = []
      commentGuideItems.value = []
      replyTemplateItems.value = []
      geoItems.value = []
      seoScore.value = null
    }
    seoChecks.value = []
  }

  /** 构建发布状态列表 */
  function buildPlatformStatuses(): void {
    const platforms: Platform[] = ['xiaohongshu', 'douyin', 'weixin']
    platformStatuses.value = platforms.map(p => {
      const record = publishRecordsMap.value.get(p)
      return {
        platform: p,
        record_id: record?.id ?? '',
        status: record?.publish_status ?? 'unpublished',
        published_at: record?.published_at ?? null
      }
    })
  }

  /** 切换平台，溯源：数据流-状态流转"平台切换状态流转" */
  async function switchPlatform(platform: Platform): Promise<void> {
    if (platform === currentPlatform.value) return

    // 自动保存当前平台修改
    if (hasUnsavedChanges.value && currentRecordId.value) {
      await publishService.autoSave(currentRecordId.value, {
        title: platformTitle.value,
        description: platformDescription.value,
        tags: tags.value
      })
      hasUnsavedChanges.value = false
    }

    currentPlatform.value = platform
    loadPlatformData(platform)
  }

  /** AI生成文案（换一批），溯源：数据流-状态流转"AI生成文案状态流转" */
  async function generateCopy(): Promise<void> {
    if (!currentRecordId.value || !scriptInfo.value) return
    isGeneratingCopy.value = true
    try {
      const result = await publishService.generateCopy(
        currentRecordId.value,
        {
          platform: currentPlatform.value,
          conversion_type: conversionGoal.value,
          script_text: scriptInfo.value.full_text,
          variation: true
        }
      )
      platformTitle.value = result.title
      platformDescription.value = result.description
      tags.value = result.tags
      commentGuideItems.value = result.comment_guide.map((text, i) => ({ id: `guide-${i}`, text }))
      replyTemplateItems.value = result.reply_templates.map((text, i) => ({ id: `reply-${i}`, text }))
      hasUnsavedChanges.value = true
    } finally {
      isGeneratingCopy.value = false
    }
  }

  /** SEO检查，溯源：数据流-状态流转"SEO检查状态流转" */
  async function performSeoCheck(): Promise<void> {
    if (!currentRecordId.value) return
    isSeoChecking.value = true
    try {
      const result = await publishService.seoCheck(
        currentRecordId.value,
        {
          title: platformTitle.value,
          description: platformDescription.value,
          tags: tags.value,
          platform: currentPlatform.value,
          conversion_type: conversionGoal.value
        }
      )
      seoScore.value = result.seo_score
      seoChecks.value = result.checks

      // 更新publishRecordsMap中的seo_score
      const record = publishRecordsMap.value.get(currentPlatform.value)
      if (record) {
        record.seo_score = result.seo_score
      }
    } finally {
      isSeoChecking.value = false
    }
  }

  /** 添加话题标签，溯源：数据流-状态流转"话题标签操作状态流转" */
  async function addTag(tagText: string): Promise<void> {
    if (isTagLimitReached.value) return
    if (!currentRecordId.value) return

    const result = await publishService.addTag(currentRecordId.value, { tag: tagText })
    tags.value = result.tags
    hasUnsavedChanges.value = true
  }

  /** 删除话题标签 */
  async function removeTag(tagText: string): Promise<void> {
    if (!currentRecordId.value) return

    const result = await publishService.removeTag(currentRecordId.value, tagText)
    tags.value = result.tags
    hasUnsavedChanges.value = true
  }

  /** 生成GEO图文，溯源：数据流-状态流转"GEO图文生成状态流转" */
  async function generateGeo(): Promise<void> {
    if (!currentRecordId.value || !scriptInfo.value) return
    isGeneratingGeo.value = true
    try {
      const result = await publishService.generateGeo(
        currentRecordId.value,
        {
          script_text: scriptInfo.value.full_text,
          tags: tags.value
        }
      )
      geoItems.value = result.geo_items
    } finally {
      isGeneratingGeo.value = false
    }
  }

  /** 标记已发布，溯源：数据流-状态流转"标记已发布状态流转" */
  async function markPublished(recordId: string): Promise<void> {
    // 检查SEO评分
    if (seoScore.value !== null && seoScore.value < 40) {
      return // 按钮已禁用，此处为二次校验
    }

    // 检查AIGC确认
    if (!aigcConfirmed.value) {
      // 弹出AIGC确认弹窗，由组件层处理
      return
    }

    const result = await publishService.markPublished(recordId, {
      aigc_confirmed: aigcConfirmed.value
    })

    // 更新本地数据
    const record = publishRecordsMap.value.get(currentPlatform.value)
    if (record) {
      record.publish_status = 'published'
      record.published_at = new Date().toISOString()
    }
    buildPlatformStatuses()
  }

  /** 确认AIGC标识 */
  async function confirmAigc(): Promise<void> {
    if (!currentRecordId.value) return
    await publishService.confirmAigc(currentRecordId.value)
    aigcConfirmed.value = true
  }

  /** 触发自动保存，溯源：数据流-状态流转"文案自动保存状态流转" */
  function triggerAutoSave(): void {
    hasUnsavedChanges.value = true
    if (autoSaveTimer) clearTimeout(autoSaveTimer)
    autoSaveTimer = setTimeout(async () => {
      if (!currentRecordId.value || !hasUnsavedChanges.value) return
      try {
        await publishService.autoSave(currentRecordId.value, {
          title: platformTitle.value,
          description: platformDescription.value,
          tags: tags.value
        })
        hasUnsavedChanges.value = false
      } catch {
        // 自动保存失败不提示，下次修改时重试
      }
    }, 2000) // 防抖2秒
  }

  /** 保存转化目标 */
  async function saveConversionGoal(type: ConversionGoal): Promise<void> {
    conversionGoal.value = type
    if (!currentRecordId.value) return
    await publishService.saveConversionType(currentRecordId.value, {
      conversion_type: type
    })
  }

  return {
    // state
    videoProductId, video, scriptInfo, publishRecordsMap,
    currentPlatform, conversionGoal, aigcConfirmed,
    platformTitle, platformDescription, tags,
    seoScore, seoChecks, isSeoChecking,
    commentTab, commentGuideItems, replyTemplateItems,
    geoItems, isGeneratingGeo,
    platformStatuses, isGeneratingCopy,
    hasUnsavedChanges, loading,
    // computed
    currentRecordId, titleLength, descLength,
    tagLimit, isTagLimitReached, canPublish,
    // actions
    initPage, switchPlatform, loadPlatformData,
    generateCopy, performSeoCheck,
    addTag, removeTag,
    generateGeo, markPublished, confirmAigc,
    triggerAutoSave, saveConversionGoal
  }
})
```

### SEO评分算法实现（后端）

```typescript
/**
 * SEO评分算法，溯源：规则7
 * 5维度加权计算：
 * - 关键词覆盖 30%
 * - 标题质量 25%
 * - 正文结构 20%
 * - 互动引导 15%
 * - 标签使用 10%
 */
function calculateSeoScore(params: {
  title: string
  description: string
  tags: string[]
  platform: Platform
  conversion_type: ConversionGoal
}): { score: number; checks: SeoCheck[] } {
  const checks: SeoCheck[] = []

  // 1. 关键词覆盖（30%）
  const keywordScore = evaluateKeywordCoverage(params.title, params.description, params.tags)
  checks.push({
    name: '关键词覆盖',
    status: keywordScore >= 60 ? 'pass' : keywordScore >= 40 ? 'warning' : 'fail',
    score: keywordScore,
    weight: 30,
    suggestion: keywordScore < 60 ? '建议在标题和描述中增加核心关键词' : undefined
  })

  // 2. 标题质量（25%）
  const titleScore = evaluateTitleQuality(params.title, params.platform)
  checks.push({
    name: '标题质量',
    status: titleScore >= 60 ? 'pass' : titleScore >= 40 ? 'warning' : 'fail',
    score: titleScore,
    weight: 25,
    suggestion: titleScore < 60 ? '建议标题包含数字、疑问句或情感词' : undefined
  })

  // 3. 正文结构（20%）
  const structureScore = evaluateContentStructure(params.description)
  checks.push({
    name: '正文结构',
    status: structureScore >= 60 ? 'pass' : structureScore >= 40 ? 'warning' : 'fail',
    score: structureScore,
    weight: 20,
    suggestion: structureScore < 60 ? '建议使用分段、列表和emoji增强可读性' : undefined
  })

  // 4. 互动引导（15%）
  const interactionScore = evaluateInteractionGuide(params.description)
  checks.push({
    name: '互动引导',
    status: interactionScore >= 60 ? 'pass' : interactionScore >= 40 ? 'warning' : 'fail',
    score: interactionScore,
    weight: 15,
    suggestion: interactionScore < 60 ? '建议添加提问、投票或@好友等互动引导' : undefined
  })

  // 5. 标签使用（10%）
  const tagScore = evaluateTagUsage(params.tags, params.platform)
  checks.push({
    name: '标签使用',
    status: tagScore >= 60 ? 'pass' : tagScore >= 40 ? 'warning' : 'fail',
    score: tagScore,
    weight: 10,
    suggestion: tagScore < 60 ? '建议增加热门标签和长尾标签的组合' : undefined
  })

  // 加权总分
  const score = Math.round(
    (keywordScore * 0.30) +
    (titleScore * 0.25) +
    (structureScore * 0.20) +
    (interactionScore * 0.15) +
    (tagScore * 0.10)
  )

  return { score, checks }
}
```
