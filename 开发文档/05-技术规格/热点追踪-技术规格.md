## 2. 组件 Props 定义

### 2.1 TypeScript 数据类型

```typescript
/** 来源平台枚举 */
type SourcePlatform = 'douyin' | 'xiaohongshu' | 'weibo' | 'travel_media' | '5118' | 'manual';

/** 热度趋势枚举 */
type HeatTrend = 'up' | 'stable' | 'down';

/** 相关度级别枚举 */
type RelevanceLevel = 'high' | 'medium' | 'low';

/** 使用状态枚举 */
type UsageStatus = 'unused' | 'used' | 'expired';

/** 有效期状态 */
type ValidityStatus = 'urgent' | 'limited' | 'evergreen';

/** 热点列表项 */
interface HotspotItem {
  id: number;                          // 📥存储 热点信息.id
  title: string;                       // 📥存储 热点信息.title
  source_platform: SourcePlatform;     // 📥存储 热点信息.source_platform
  heat_value: number;                  // 📥存储 热点信息.heat_value
  heat_trend: HeatTrend;               // 📥存储 热点信息.heat_trend
  relevance_score: number;             // 📥存储 热点信息.relevance_score（0-100）
  ai_suggestion: string;               // 📥存储 热点信息.ai_suggestion
  keywords: string[];                  // 📥存储 热点信息.keywords
  published_at: string;                // 📥存储 热点信息.published_at，格式 ISO 8601
  valid_until: string;                 // 📥存储 热点信息.valid_until，格式 ISO 8601
  usage_status: UsageStatus;           // 📥存储 热点信息.usage_status
}

/** 热点列表API响应 */
interface HotspotListResponse {
  total: number;                       // 📊计算 热点总数
  page: number;
  page_size: number;
  items: HotspotItem[];
}

/** 监控关键词项 */
interface KeywordMonitorItem {
  id: number;                          // 📥存储 关键词监控.id
  keyword: string;                     // 📥存储 关键词监控.keyword
  match_count: number;                 // 📥存储 关键词监控.match_count
  last_matched_at: string | null;      // 📥存储 关键词监控.last_matched_at
  created_at: string;                  // 📥存储 关键词监控.created_at
}

/** 关键词列表API响应 */
interface KeywordListResponse {
  items: KeywordMonitorItem[];
}

/** 趋势数据点 */
interface TrendDataPoint {
  date: string;                        // 日期 YYYY-MM-DD
  match_count: number;                 // 匹配数量
}

/** 关键词趋势项 */
interface KeywordTrendItem {
  keyword: string;                     // 关键词
  trend_data: TrendDataPoint[];        // 趋势数据
}

/** 关键词趋势API响应 */
interface KeywordTrendResponse {
  items: KeywordTrendItem[];
}

/** 筛选条件 */
interface HotspotFilterState {
  source_platform: string;             // 📥存储 'all' | SourcePlatform
  time_range: string;                  // 📥存储 'today' | '24h' | '7d'
  relevance: string;                   // 📥存储 'all' | RelevanceLevel
  keyword: string;                     // 📥存储 搜索关键词
}

/** 排序条件 */
interface HotspotSortState {
  sort_by: string;                     // 📥存储 heat_value/relevance_score/published_at
  order: 'asc' | 'desc';              // 📥存储 排序方向
}

/** 刷新热点响应 */
interface RefreshHotspotResponse {
  success: boolean;
  new_count: number;                   // 新增热点数量
}

/** 手动录入热点请求 */
interface ManualAddHotspotRequest {
  title: string;                       // 热点标题（必填）
  source_platform: SourcePlatform;     // 来源平台（必填）
  source_url?: string;                 // 热点链接（可选）
  note?: string;                       // 备注（可选）
}

/** 手动录入热点响应 */
interface ManualAddHotspotResponse {
  success: boolean;
  id: number;                          // 新建热点ID
}

/** 标记过期响应 */
interface ExpireHotspotResponse {
  success: boolean;
}

/** 添加监控关键词请求 */
interface AddKeywordRequest {
  keyword: string;                     // 关键词（必填，唯一）
}

/** 添加监控关键词响应 */
interface AddKeywordResponse {
  success: boolean;
  id: number;                          // 新建关键词ID
}

/** 匹配提醒数据 */
interface MatchAlert {
  keyword: string;                     // 📥存储 匹配到的关键词
  match_count: number;                 // 📥存储 新匹配数量
  matched_hotspots: string[];          // 📥存储 匹配到的热点标题（最多3条）
}
```

## 3. API 服务函数

### 3.1 查询类

```typescript
/**
 * 获取热点列表
 * GET /api/v1/hotspots
 */
function getHotspots(params: {
  page?: number;                       // 默认 1
  page_size?: number;                  // 默认 20
  source_platform?: SourcePlatform;    // 可选
  time_range?: string;                 // 可选：today/24h/7d
  relevance?: RelevanceLevel;          // 可选：high/medium/low
  keyword?: string;                    // 可选：按标题模糊搜索
  sort_by?: string;                    // 默认 heat_value：heat_value/relevance_score/published_at
  order?: 'asc' | 'desc';             // 默认 desc
}): Promise<HotspotListResponse>

/**
 * 获取监控关键词列表
 * GET /api/v1/keyword-monitors
 */
function getKeywordMonitors(): Promise<KeywordListResponse>

/**
 * 获取关键词趋势
 * GET /api/v1/keyword-monitors/trends
 */
function getKeywordTrends(params: {
  keyword_id?: number;                 // 可选，不传则返回全部关键词趋势
  days?: number;                       // 默认 7
}): Promise<KeywordTrendResponse>
```

### 3.2 操作类

```typescript
/**
 * 刷新热点（触发即时采集）
 * POST /api/v1/hotspots/refresh
 */
function refreshHotspots(data?: {
  source_platform?: SourcePlatform;    // 可选，指定刷新来源
}): Promise<RefreshHotspotResponse>

/**
 * 手动录入热点
 * POST /api/v1/hotspots
 */
function createHotspot(data: ManualAddHotspotRequest): Promise<ManualAddHotspotResponse>

/**
 * 标记热点过期
 * PUT /api/v1/hotspots/:id/expire
 */
function expireHotspot(id: number): Promise<ExpireHotspotResponse>

/**
 * 添加监控关键词
 * POST /api/v1/keyword-monitors
 */
function addKeywordMonitor(data: AddKeywordRequest): Promise<AddKeywordResponse>

/**
 * 删除监控关键词
 * DELETE /api/v1/keyword-monitors/:id
 */
function deleteKeywordMonitor(id: number): Promise<{ success: boolean }>
```

## 4. 数据库表结构

### 4.1 热点信息表 (hotspots)

| 字段名 | 类型 | 约束 | 说明 | 溯源 |
|--------|------|------|------|------|
| id | SERIAL | PRIMARY KEY | 主键 | 📥存储 |
| title | VARCHAR(500) | NOT NULL | 热点标题 | 📥存储 |
| source_platform | VARCHAR(30) | NOT NULL, INDEX | 来源平台：douyin/xiaohongshu/weibo/travel_media/5118/manual | 📥存储 |
| source_url | VARCHAR(1000) | | 来源链接 | 📥存储 |
| heat_value | INTEGER | DEFAULT 0, INDEX | 热度值 | 📥存储 |
| heat_trend | VARCHAR(10) | DEFAULT 'stable' | 热度趋势：up/stable/down | 📥存储 |
| relevance_score | INTEGER | DEFAULT 0, INDEX | 相关度评分（0-100） | 📥存储 |
| ai_suggestion | TEXT | | AI结合建议 | 📥存储 |
| keywords | TEXT[] | | 关键词标签数组 | 📥存储 |
| published_at | TIMESTAMP WITH TIME ZONE, INDEX | | 发布时间 | 📥存储 |
| valid_until | TIMESTAMP WITH TIME ZONE | | 有效期截止时间 | 📥存储 |
| usage_status | VARCHAR(20) | DEFAULT 'unused', INDEX | 使用状态：unused/used/expired | 📥存储 |
| note | TEXT | | 备注（手动录入用） | 📥存储 |
| created_at | TIMESTAMP WITH TIME ZONE | DEFAULT NOW() | 创建时间 | 📥存储 |
| updated_at | TIMESTAMP WITH TIME ZONE | DEFAULT NOW() | 更新时间 | 📥存储 |

**索引清单**：
- `idx_hotspots_source_platform` ON (source_platform) — 来源筛选
- `idx_hotspots_published_at` ON (published_at) — 时间筛选+排序
- `idx_hotspots_heat_value` ON (heat_value) — 热度排序
- `idx_hotspots_relevance_score` ON (relevance_score) — 相关度排序
- `idx_hotspots_usage_status` ON (usage_status) — 使用状态筛选

### 4.2 关键词监控表 (keyword_monitors)

| 字段名 | 类型 | 约束 | 说明 | 溯源 |
|--------|------|------|------|------|
| id | SERIAL | PRIMARY KEY | 主键 | 📥存储 |
| keyword | VARCHAR(100) | NOT NULL, UNIQUE | 监控关键词（唯一） | 📥存储 |
| match_count | INTEGER | DEFAULT 0 | 匹配数量 | 📥存储 |
| last_matched_at | TIMESTAMP WITH TIME ZONE | | 最后匹配时间 | 📥存储 |
| created_at | TIMESTAMP WITH TIME ZONE | DEFAULT NOW() | 创建时间 | 📥存储 |
| updated_at | TIMESTAMP WITH TIME ZONE | DEFAULT NOW() | 更新时间 | 📥存储 |

**约束清单**：
- `uk_keyword_monitors_keyword` UNIQUE ON (keyword) — 关键词唯一约束

## 5. 状态管理

### 5.1 数据归属

| 数据 | 存储位置 | 类型 | 说明 |
|------|---------|------|------|
| 热点列表 (hotspot_list) | 组件状态 ref | `HotspotItem[]` | 页面加载/筛选/搜索/翻页/刷新时更新 |
| 分页信息 (page, page_size, total) | 组件状态 ref | `{ page: number, page_size: number, total: number }` | 分页加载 |
| 筛选条件 (filters) | 组件状态 reactive | `HotspotFilterState` | 默认：all/today/all/'' |
| 排序条件 (sortState) | 组件状态 ref | `HotspotSortState` | 默认：heat_value/desc |
| 搜索关键词 (search_keyword) | 组件状态 ref | `string` | 防抖500ms后触发筛选 |
| 监控关键词列表 (keywords) | 组件状态 ref | `KeywordMonitorItem[]` | 页面加载时获取，添加/删除后刷新 |
| 趋势曲线数据 (keyword_trends) | 组件状态 ref | `KeywordTrendItem[]` | 页面加载+关键词变更后刷新 |
| 展开的建议卡片ID (expanded_suggestion_id) | 组件状态 ref | `number \| null` | 默认 null，点击"查看建议"切换 |
| 加载状态 (loading) | 组件状态 ref | `boolean` | 默认 true |
| 刷新中 (refreshing) | 组件状态 ref | `boolean` | 默认 false |
| 手动录入弹窗可见 (show_add_modal) | 组件状态 ref | `boolean` | 默认 false |
| 录入表单数据 (add_form) | 组件状态 reactive | `ManualAddHotspotRequest` | 弹窗打开时初始化空表单 |
| 新匹配提醒弹窗 (match_alert) | 组件状态 ref | `MatchAlert \| null` | 默认 null |

### 5.2 计算属性

| 属性名 | 类型 | 计算逻辑 | 说明 |
|--------|------|---------|------|
| hotspot_total | `number` | `total`（来自API响应） | 📊计算 热点总数 |

### 5.3 前端格式化函数

| 函数名 | 输入 | 输出 | 说明 |
|--------|------|------|------|
| formatHeatValue | `number` | `string` | 📊计算 ≥10000显示"X.X万"；>100万显示"100万+" |
| computeValidity | `valid_until: string` | `ValidityStatus` | 📊计算 ≤24h→urgent, ≤48h→limited, >48h→evergreen |
| mapSourcePlatform | `SourcePlatform` | `string` | 📥存储 映射为中文显示名 |
| mapHeatTrend | `HeatTrend` | `{ icon: string, color: string }` | 📥存储 映射为图标和颜色 |

### 5.4 加载策略

| 时机 | 触发 | API调用 | 备注 |
|------|------|---------|------|
| 页面初始加载 | `onMounted` | `getHotspots()` + `getKeywordMonitors()` + `getKeywordTrends()` | 并行请求 |
| 筛选变更 | watch filters | `getHotspots()` | 重置 page=1 |
| 排序变更 | watch sortState | `getHotspots()` | 重置 page=1 |
| 搜索输入 | watch keyword (debounce 500ms) | `getHotspots()` | 重置 page=1 |
| 刷新热点 | 按钮点击 | `refreshHotspots()` → `getHotspots()` | 先刷新，再重载列表 |
| 手动录入 | 表单提交 | `createHotspot(data)` | 成功后插入新卡片到列表顶部 |
| 标记过期 | 按钮点击 | `expireHotspot(id)` | 成功后更新卡片状态 |
| 添加关键词 | 输入确认 | `addKeywordMonitor(data)` → `getKeywordMonitors()` + `getKeywordTrends()` | 刷新关键词列表和趋势 |
| 删除关键词 | 点击删除 | `deleteKeywordMonitor(id)` → `getKeywordMonitors()` + `getKeywordTrends()` | 刷新关键词列表和趋势 |
| 匹配提醒 | WebSocket/轮询 | 系统触发，无需前端主动调用 | 弹窗显示 |
